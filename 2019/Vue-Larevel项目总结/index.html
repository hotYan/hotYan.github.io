二、流程逻辑

2.1 数据表

  1. 设计表（略）

  2. 创建建表文件

    $ php artisan make:migrate create_hotyan_init_table --create=hotyan_init
#以上命令会创建migrate文件，在database/migrations目录下生成新文件：date_create_hotyan_init_table.php

  3. 编辑建表文件

    public function up()
{
    <!-- 新增表：字段类型、字段修饰 -->
    Schema::create('hotyan_table', function (Blueprint $table) {
        $table->increments('hotyan_id')->comment('自增主键');
        $table->string('hotyan_name',25)->nullable()->comment('名字');
        $table->integer('hotyan_age')->default(10)->comment('年龄');
        $table->date('created_at');
        $table->timestamps();
    });
    <!-- 修改表 -->
    Schema::table('hotyan_table', function (Blueprint $table) {
        $table->string('hotyan_sex')->comment('性别');     //新增sex
        $table->integer('hotyan_grade')->comment('成绩');  //新增grade
        $table->string('hotyan_name',50)->nullable()->comment('名字')->change();   //修改长度
    });
}
public function down()
{
    Schema::dropIfExists('hotyan_table');//删除
    Schema::table('hotyan_table', function (Blueprint $table) {
        $table->dropColumn(['hotyan_sex','hotyan_grade']);    //删除sex，grade
        $table->string('hotyan_name',25)->nullable()->comment('名字')->change();   //修改之前
    });
}

  1. 生成数据表

    $ php artisan migrate   //数据库将新增hotyan_table表

            </el-select>
        </el-form-item>
    </el-form>
    <!--弹窗footer-->
    <span slot="footer" class="dialog-footer">

<script>
    export default{
        data(){
            let checkName(rule,value,callback){
                let reg = /^[\u4E00-\u9FA5A-Za-z0-9_]+$/;
                if (reg.exec(value)) {
                    callback(new Error("只含中文、数字、字母、下划线"));
                } else {
                    callback();
                }
            };
            return{
                dialogVisible: false,
                createFormLoading: false
                createForm: {
                    name: '',
                    sex: ''
                },
                createFormRules:{
                    name: [
                        {required: true,trigger: "blur",message: "请输入用户名"},
                        {pattern: /^[\u4E00-\u9FA5A-Za-z0-9_]+$/,trigger: "blur",message: "只含中文、数字、字母、下划线"},
                        {validator: checkName,trigger: "blur",message:"只含中文、数字、字母、下划线"}],
                    sex: [{required: true,message: "请选择性别",trigger: "change"}]
                },
            }
        },
        methods: {
            cancel(){
                <!-- 关闭弹窗并清空表单 -->
                this.dialogVisible = false;     //关闭弹窗
                this.createForm = {     //置空表单
                    name: '',
                    sex: ''
                };
            },
            submit(formName){
                this.createFormLoading = true;  //开启等待状态，加载中
                this.$refs[formName].validate(valid => {
                    if (!valid){
                        return (this.createFormLoading = false);
                    }  
                    <!--发起存数据请求-->
                    this.$api.hotyanApi.create("hotyan",this[formName]).then(res => {
                        this.$message.success("添加成功");
                        this.createFormLoading = false;  //关闭等待状态，加载中
                        this.cancel();      //关闭弹窗，清空表单
                        this.getList();     //获取数据列表
                    })
                    .catch(err => {})
                    .finally(() => (this.createFormLoading = false));
                })
            }
        }
    }
</script>


3.2 获取用户列表

<!-- Table表格 -->
<el-table :data="tableData" border  style="width: 100%" v-loading="tableLoading" @selection-change="handleSelectionChange" ref="multipleTable">
    <el-table-column type="selection" width="55" > </el-table-column>
    <el-table-column prop="name"  label="姓名" width="180"> </el-table-column>
    <el-table-column prop="sex"  label="性别" width="180"> </el-table-column>
    <el-table-column prop="age"  label="年龄" width="180"> </el-table-column>
    <el-table-column prop="geade"  label="分数" width="180"> </el-table-column>
</el-table>
<script>
    export default{
        data(){
            return{
                pageInfo: {
                    page: 1,
                    per_page: 10,
                    totalNum: 0,
                    name: "",
                    start_date: '',
                    end_date:''
                },
                tableData:[],
                tableLoading: false ,
                multipleSelection: []
            }
        },
        methods: {
            cancel(){
                <!-- 见3.1 -->
            },
            submit(formName){
                <!-- 见3.1 -->
            },
            getList(){
                this.tableLoading = true;
                _this.tableData=[]; //置空上一次数据
                let _this = this;
                var data = [];
                this.$api.hotyanApi.list("hotyan",this.pageInfo).then(res => {
                    for(let i = 0; i < res.data.length; i++) {
                        var obj = {};
                        obj.name = res.data[i].name;
                        obj.sex = res.data[i].sex;
                        obj.age = res.data[i].age;
                        obj.grade = res.data[i].grade;
                        data[i] = obj;
                        _this.tableData.push(obj);
                    }
                    this.pageInfo.page = res.meta.pagination.current_page;
                    this.pageInfo.per_page = res.meta.pagination.per_page;
                    this.pageInfo.totalNum = res.meta.pagination.total;
                    this.tableLoading = false;
                })
                .catch(err => {})
                .finally(() => (this.tableLoading = false));
            },
            handleSelectionChange(val) {
                this.multipleSelection = val;
            }
        }
    }
    </script>

    const hotyanApi = {
    list(uri, params = {}){
        return axios.get('/api/' + uri + '/', { params: params });
    },
    create(uri, params){
        return axios.post('/api/' + uri + '/', params);
    },
    edit(uri, id, params){
        return axios.put('/api/' + uri + '/' + id, params );
    },
    delete(uri, id){
        return axios.delete('/api/' + uri + '/' + id);
    },
    getDetails(uri, id){
        return axios.get('/api/' + uri + '/' ,{ params:{ id : id } });
    }
};
export default hotyanApi;
